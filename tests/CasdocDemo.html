    package ca.mcgill.cs.stl.casdoc;

import java.nio.charset.StandardCharsets;
import java.util.Base64;
import java.util.Base64.Decoder;
import java.util.Base64.Encoder;

/**
 * Example showing the main features of Casdoc.
 * This code shows how to encode and decode a message using Base64.
 */
public class CasdocDemo
{
	public static void main(String[] args)
	{

    		String message = "Hello, World!";
		byte[] input = message.getBytes(StandardCharsets.UTF_8);

    		Encoder encoder = Base64.getEncoder();
		String base64 = encoder.encodeToString(input);
    
		System.out.println(base64);
    
		

    		Decoder decoder = Base64.getDecoder();
		byte[] output = decoder.decode(base64);
    
		String decoded = new String(output, StandardCharsets.UTF_8);
		System.out.println(decoded);
    
	}
}
